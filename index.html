function calcVO2() { let distance = parseFloat(document.getElementById("distance").value); let time = parseFloat(document.getElementById("timeMin").value); if (!distance || !time) { document.getElementById("vo2Result").innerText = "Remplis tous les champs"; return; } let speed = (distance * 1000) / time; let vo2 = 0.2 * speed + 3.5; document.getElementById("vo2Result").innerText = "VO2 max estimé : " + vo2.toFixed(1) + " ml/kg/min"; }
